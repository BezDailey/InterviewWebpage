15. Insert into mobile media query layout brackets 

    /* Takes Away Flexbox */
    body section {
        flex: none;
    }
    
    /* Stretches Body Element */
    body {
        width: 100%;
    }

    /* Stops Displaying Navigation Menu For Tablet And Desktop Screens */
    section nav {
        display: none;
    }

    /* Displays Navicon Icon */
    a#navicon {
        display: block;
    }

    /* Stops Displaying Mobile Navigation Menu */
    body > nav ul {
        display: none;
        margin-block-start: 0px;
        margin-block-end: 0px;
        background-color: #C0C0C0;
    }

    /*Displays Mobile Navigation Menu When Using Navicon */
    a#navicon:hover+ul, nav.horizontal ul:hover {
        display: block;
    }

    /*Changes Text-Align For Mobile Navigation Menu to Left */
    body nav ul {
        text-align: left;
    }

    /* Adds Article Margin */
    article {
        margin: 0px 2% 30px 2%;
    }

16. Insert into tablet styles media query brackets

    /*Makes Min-Height For Body Element 100% */
    body {
        min-height: 100%;
    }

    /* Stops showing the Mobile Navigation menu */
    body > nav {
        display: none;
    }
